comment:4 Changed 9 years ago by Janik Zikovsky

Sample script for testing: 

CreateMDWorkspace(Dimensions='3',Extents='0,10,0,10,0,10',Names='x,y,z',Units='m,m,m',SplitInto='5',MaxRecursionDepth='20',OutputWorkspace='mdw')
FakeMDEventData("mdw",  PeakParams="1e6, 5,5,5, 2")
BinToMDHistoWorkspace("mdw", "A",  1, "x,0,10,30", "y,0,10,30", "z,0,10,30", IterateEvents="1", Parallel="0")

CreateMDWorkspace(Dimensions='3',Extents='0,10,0,10,0,10',Names='x,y,z',Units='m,m,m',SplitInto='5',MaxRecursionDepth='20',OutputWorkspace='mdw')
FakeMDEventData("mdw",  PeakParams="1e6, 3,3,3, 3")
BinToMDHistoWorkspace("mdw", "B",  1, "x,0,10,30", "y,0,10,30", "z,0,10,30", IterateEvents="1", Parallel="0")

A = mtd['A']
B = mtd['B']

C = A > 0
D = B > 0

C_or_D = C | D
C_and_D = C & D
C_xor_D = C ^ D

C_or_D2 = C
C_or_D2 |= D

C_and_D2 = C
C_and_D2 &= D

C_xor_D2 = C
C_xor_D2 ^= D

E = (C == D)
