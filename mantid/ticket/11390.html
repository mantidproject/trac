Description

According to Tim, running this twice causes Mantid to hang. Could you please try to trackdown the origin of the problem and either fix it or pass the ticket back to Owen.

This affects the 3.3 release and the nightly builds according to TIm, so we should fix it ASAP. 

import os, sys
sys.path.append(r'C:\MantidInstall')
sys.path.append(r'C:\MantidInstall\bin')
sys.path.append(r'u:\user\mantid\TestingPointDetectorReduction')
from mantid.simpleapi import *
from WrappedReduction import *
r=Runs()

## Build a transmission
trans = transmission('11711+11712+11713+11683+11633+11634+11635')
## Get the data
r1q = refq('11716+11723', theta=0.50, Transmission = trans)
r2q = refq('11718+11717+11724', theta=1.0, Transmission = trans)
r3q = refq(r[11719:11722,11725:11727], theta=2.0, Transmission = trans)

##  Rebin
rr1q = Rebin(r1q,'0.010,-0.01,0.05')
rr2q = Rebin(r2q,'0.018,-0.01,0.1')
rr3q = Rebin(r3q,'0.03,-0.01,0.25')

##  Stitch
rr,scale = Stitch1DMany('rr1q,rr2q,rr3q', Params='-0.015’)
print "Happy”
rr_2,scale_2 = Stitch1DMany('rr1q,rr2q,rr3q', Params='-0.015')
print "Happy”
rr,scale = Stitch1DMany('rr1q,rr2q,rr3q', Params='-0.015')
print “UnHappy. Execution gets stuck on the line above.”
