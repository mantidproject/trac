The correct script now is: 

import ISISCommandInterface as ici
run = '19682'
ev, mon = Load(run, LoadMonitors=True)
b = '5.5,45.5,50.0, 50.0,1000.0, 500.0,1500.0, 750.0,99750.0, 255.0,100005.0'
hist = Rebin(ev, b, False)
mon_h = Rebin(mon, b, False)
mon_name = mon_h.name()
ConjoinWorkspaces(mon_h, hist, True)
hist = RenameWorkspace(mon_name)

ici.SANS2D()
ici.MaskFile('/apps/mantid/systemtests/Data/SANS2D/MASKSANS2D.091A')
ici.AssignSample(hist,reload=False)
ici.Gravity(True)
wav1 = 2.0
wav2 = wav1 + 2.0
reduced = ici.WavRangeReduction()
event_reduced = RenameWorkspace(reduced)

current = LoadNexus(run)
ici.SANS2D()
ici.MaskFile('/apps/mantid/systemtests/Data/SANS2D/MASKSANS2D.091A')
ici.AssignSample(current,reload=False)
ici.Gravity(True)
wav1 = 2.0
wav2 = wav1 + 2.0
reduced = ici.WavRangeReduction()
histogram_reduced = RenameWorkspace(reduced)

plot(event_reduced, 0)
plot(histogram_reduced,0)

The reason, is that now, loading event, does not load the monitors spectra, and as so, can be 'conjoined' to the monitors straight forward.
