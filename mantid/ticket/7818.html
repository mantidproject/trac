To Test: Execute the following script, click "Show Instrument", and see that the detector has zebra stripes where every second tube (along y-axis) is masked. 

import mantid
from mantid.simpleapi import *
from reduction_workflow.instruments.sans.hfir_command_interface import *

BIOSANS()
SetSampleDetectorOffset(837.9)
SolidAngle(detector_tubes=True)
MonitorNormalization()
AzimuthalAverage(n_bins=100, n_subpix=1, log_binning=False)
NoIQxQy()
SetBeamCenter(96.7, 16.6)
NoSensitivityCorrection()
SetTransmission(1.0,0.0)
ThetaDependentTransmission(False)
AppendDataFile(["BioSANS_test_data.xml"])
NoSaveIq()
MaskDetectorSide('Back')
Reduce()
